SELECT p2.name, p2.the_geom, p2.classification, ST_GoogleMapLink(p2.the_geom)
FROM points p1, points p2
WHERE ST_INTERSECTS(ST_BUFFER(p1.THE_GEOM, 0.17),
p2.the_geom)
AND p1.id = 278
AND p2.id != 278
and REGEXP_LIKE(p2.classification, '^(.*,M,)|(^M,).*$|(^.*,M$)', 'c');
